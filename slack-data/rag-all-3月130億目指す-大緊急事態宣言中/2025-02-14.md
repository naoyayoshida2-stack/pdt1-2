さっきクエリ回してたら昨日の振り分けがおかしなことになってたんだけど、これ確認してもらえない？
クエリは↓
```--変数を宣言する
declare beta_start_date date default date('2024-09-13'); --11月4日週以降の実績を引いてくる
declare beta_end_date date default date('2024-12-25'); --12月26日で一旦停止、その前日までの総受を対象とする
declare start_date date default date('2025-01-30'); --29日に切り戻し、30日からの実績を集計する
declare end_date date default current_date() - 1;

--raw
with ro2_raw as (
  select
    *,
    jobseeker_id||jobseeker_branch_id as js_id,
    entry_complete_timestamp as entry_timestamp,
    date_trunc(entry_start_date,week(monday)) as entry_week
  from
  dharma-dwh-rag.datamart.v_rag_entry_users_for_ro2
  where
  (entry_start_date >= '2025-02-10')
  and test_group like "%Group%"
  and entry_start_type != "unknown"
  and entry_start_type = "independent_entry_form" -- 独立EF（982）
)

select
  pdt2_user_flag,
  entry_start_date,
  sum(1) as distribution_uu
from
ro2_raw
group by all
order by 1,2```
