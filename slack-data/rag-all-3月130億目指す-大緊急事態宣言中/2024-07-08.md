アプリDL率が何かPDT2すごい高い、これ変かな？クエリ確認してほしい
```--pdt1のAPP利用
with pdt1 as (
  select
    "pdt1" as pdt2_user_flag,
    count(distinct concat(jobseeker_id,jobseeker_branch_id)) as js_uu,
    count(distinct case when ca_interview_date is not null then concat(jobseeker_id,jobseeker_branch_id) else null end) as mendan_uu,
    count(distinct case when date(entry_timestamp) >= date(last_access_date) then concat(jobseeker_id,jobseeker_branch_id) else null end) as app_uu,
    count(distinct case when date(entry_timestamp) >= date(last_access_date) then concat(jobseeker_id,jobseeker_branch_id) else null end)
    / count(distinct concat(jobseeker_id,jobseeker_branch_id)) as entry_app_dl_rate
  from
  dharma-dwh-rag.datamart.t_rag_jobseeker_all a
  left outer join
  dharma-dwh-rag.wc_resource_new.t_pdt_app_management b
  on
  a.jobseeker_id = b.jobseeker_no
  where
  media_cd = '982'
  and date(entry_timestamp) >= '2024-07-01'
  and acceptance_flag = '1'
),


--PDT2_APP利用
pdt2_app as (
  select distinct
    jobseekerNo as jobseeker_id,
    jobseekerBranchNo as jobseeker_branch_id
  from
  dharma-dwh-rag.lettice.rag_events
  where
  date(eventDate) >= '2024-07-01'
  and device = 'app'
),

--PDT2_面談
pdt2 as (
  select
    "pdt2" as pdt2_user_flag,
    count(distinct concat(jobseeker_id,jobseeker_branch_id)) as js_uu,
    count(distinct case when first_ca_interview_date is not null then concat(jobseeker_id,jobseeker_branch_id) else null end) as mendan_uu,
    count(distinct case when b.jobseeker_id is not null then concat(jobseeker_id,jobseeker_branch_id) else null end) as app_uu,
    count(distinct case when b.jobseeker_id is not null then concat(jobseeker_id,jobseeker_branch_id) else null end)
    / count(distinct concat(jobseeker_id,jobseeker_branch_id)) as entry_app_dl_rate
  from
  `dharma-user.sandbox_rag.v_rag_ro2_entry_funnel` a
  left outer join
  pdt2_app b
  using(jobseeker_id,jobseeker_branch_id)
  where
  date(lv2_pass_datetime) >= '2024-07-01'
),

--union
app as (
  select *
  from
  pdt1
  union all
  select *
  from
  pdt2
)
select * from app
order by pdt2_user_flag```
