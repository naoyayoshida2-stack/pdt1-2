MTGメモ　<https://docs.google.com/document/d/1b22IClcuBGWPkBAw1KnAjUHm1sOsbRgg3zEcakHCib4/edit?tab=t.0>

サンプルクエリ（Job）
```DECLARE send_from DATE DEFAULT DATE("2024-04-25");
DECLARE send_to DATE DEFAULT DATE("2024-05-09");
declare target_grps array <string> default [
  'route_tthire_without_hire_qb_postday0_20240425_10',
  'route_tthire_postday0_20240425_90'
];
declare target_grp string default 'route_tthire_postday0_20240425_90';

with ablog as(
select *,
cast(job_id as int64)as job_id_int,
_PARTITIONDATE as partition_id
from `dharma-ml-prd.i2a_rag.ab_log`
where _PARTITIONDATE BETWEEN send_from AND send_to
and `group` IN UNNEST(target_grps)),

  job_daily_bo AS (　　--溝口さんへ　ここから
    SELECT
      partition_id,
      job_id,
      company_cd,
      rag_client_segment,
      to_be_hired_count AS bo_cnt
    FROM
      `dharma-dwh-rag.datamart.t_rag_job_pool`
    WHERE
      -- 仮説検証のため2024年以降のデータを取得
      partition_id BETWEEN send_from AND send_to
  ),

  -- EPZ企業のリスト
  stg_epz AS (
    SELECT
      corp_code AS company_cd,
      1 AS epz_flag
    FROM
      `dharma-dwh-rds.google_sheets.epz_list_240701`
    GROUP BY
      1
  ),

  -- 日毎の求人情報にBO区分とクライアントセグメントを付与
  job_with_category AS (
    SELECT
      t1.partition_id,
      t1.job_id,
      t1.company_cd,
      -- BO数に基づいたカテゴリ分け
      CASE
        WHEN t1.bo_cnt = 1 THEN '01. bo=1'
        WHEN t1.bo_cnt BETWEEN 2 AND 5 THEN '02. bo=2-4'
        WHEN t1.bo_cnt >= 5 THEN '03. bo>=5'
        ELSE '04. others'
      END AS bo_category,
      -- 顧客セグメントの定義
      CASE
        WHEN t2.epz_flag IS NOT NULL THEN "a.EPZ"
        WHEN t2.epz_flag IS NULL AND t1.rag_client_segment = "B" THEN "b.NotEPZ_B"
        WHEN t2.epz_flag IS NULL AND t1.rag_client_segment IN ("M", "R") THEN "c.NotEPZ_MR"
        ELSE "d.others"
      END AS client_segment,
  
      t1.bo_cnt
    FROM
      job_daily_bo AS t1
      LEFT JOIN stg_epz AS t2 ON t1.company_cd = t2.company_cd
  ),
  job_t as(
    select distinct
    partition_id,job_id,company_cd,client_segment,bo_category,bo_cnt
    from job_with_category),　--溝口さんへ　ここまで

  ablog_job as(
    select 
      ablog.*,
      job_t.company_cd,
      job_t.client_segment,
      job_t.bo_category,
      job_t.bo_cnt
    from ablog 
    left join job_t 
    on ablog.job_id_int=job_t.job_id
    and ablog.partition_id=job_t.partition_id
  )

select 
`group`,
client_segment,
bo_category,
count(1)as sendcnt,
count(distinct jobseeker_id)as jsuu,
count(distinct job_id)as jobuu,
from ablog_job
where has_sent=1
group by 1,2,3
order by 1,2,3```
