クエリは以下です
```/*
7月1日 α開始(1%)
7月10日 比率UP1%→2%
9月2日 Lv.3障害発生に伴い新規集客停止
9月12日 13:15:00 集客再開(19%)
9月23日 12:11:00 23%へ拡大
9月30日 11:30:00 28%へ拡大
*/

--変数を宣言する
declare start_datetime_alpha datetime default datetime('2024-08-08 10:00:00'); --evidentlyログ取得開始日
declare start_datetime_beta datetime default datetime('2024-09-12 13:15:00'); --集客再開日
declare end_date date default current_date() - 1;

--raw
with raw as (
  select
    *,
    jobseeker_id||jobseeker_branch_id as js_id,
    entry_complete_timestamp as entry_timestamp,
    date_trunc(entry_complete_timestamp,week(monday)) as entry_week,
    date(entry_complete_timestamp) as entry_date
  from
  dharma-dwh-rag.datamart.v_rag_entry_users_for_ro2
  where
  jobseeker_id is not null
  and entry_complete_timestamp >= start_datetime_alpha
  and test_group != 'Others'
),

--面談予約実績
ca_interview_reservation as (
  select
    jobseeker_id,
    jobseeker_branch_id,
    jobseeker_id||jobseeker_branch_id as js_id,
    min(interview_reservation_first_registration_datetime) as min_set_datetime,
  from
  dharma-dwh-rag.datamart.t_rag_canvas_interview_reservation
  where
  partition_id = end_date
  group by 1,2,3
),


js_raw as (
  select
    a.*,
    min_set_datetime,
  from
  raw a
  left outer join
  ca_interview_reservation
  using(js_id)
),

output as (
  select
    entry_date,
    pdt2_user_flag,
    sum(1) as entry_uu,
    sum(case when min_set_datetime is not null then 1 else 0 end) as mendan_set_uu,
    sum(case when ca_interview_date <= end_date then 1 else 0 end) as mendan_uu,
    sum(case when doc_send_count >= 1 then 1 else 0 end) as doc_send_uu,
    sum(doc_send_count) as doc_send_count,
    sum(case when view_job_count >= 1 then 1 else 0 end) as view_job_uu,
    sum(view_job_count) as view_job_count,
    sum(case when apply_count >= 1 then 1 else 0 end) as apply_uu,
    sum(apply_count) as apply_count,
    sum(case when ca_recommend_count >= 1 then 1 else 0 end) as ca_rec_uu,
    sum(ca_recommend_count) as ca_rec_count,
    sum(case when ra_recommend_count >= 1 then 1 else 0 end) as ra_rec_uu,
    sum(ra_recommend_count) as ra_rec_count,
    sum(case when doc_pass_count >= 1 then 1 else 0 end) as doc_pass_uu,
    sum(doc_pass_count) as doc_pass_count,
    sum(case when interview_count >= 1 then 1 else 0 end) as so_uu,
    sum(interview_count) as so_count,
    sum(case when informal_job_offer_count >= 1 then 1 else 0 end) as informal_job_offer_uu,
    sum(informal_job_offer_count) as informal_job_offer_count,
    sum(decide_f) as decide_uu
  from
  js_raw
  group by 1,2
)

select * from output
order by 1,2```
